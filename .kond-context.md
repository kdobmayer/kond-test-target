### Controller Pattern (`src/controllers/products.js`)
```javascript
function getProduct(req, res) {
  const db = getDb();
  const product = db.prepare('SELECT * FROM products WHERE id = ?').get(req.params.id);
  if (!product) {
    return res.status(404).json({
      error: 'Not Found',
      message: `Product with id ${req.params.id} not found`,
      timestamp: new Date().toISOString()
    });
  }
  res.json({ data: product });
}

function createProduct(req, res) {
  const db = getDb();
  const body = req.body;
  // Validation usage
  const requiredCheck = validateRequired(['name', 'sku', 'price'], body);
  if (!requiredCheck.valid) return res.status(400).json({ error: 'Validation Error', message: requiredCheck.error });
  
  try {
    const result = db.prepare('INSERT INTO products (...) VALUES (...)').run(...);
    const product = db.prepare('SELECT * FROM products WHERE id = ?').get(result.lastInsertRowid);
    res.status(201).json({ data: product });
  } catch (err) {
    if (err.code === 'SQLITE_CONSTRAINT_UNIQUE') {
      return res.status(409).json({ error: 'Conflict', message: '...', timestamp: new Date().toISOString() });
    }
    throw err;
  }
}
```

## 6. Constraints and Gotchas
- **Database:** Use `better-sqlite3`. Ensure `foreign_keys = ON`.
- **Naming:** `camelCase` for JavaScript, `snake_case` for database columns.
- **Errors:** Manually format 404, 400, and 409 errors with a `timestamp` field (ISO string) to match existing controllers.
- **Stock:** Stock reservation must be atomic (decrement on item add, increment on cancel). Use `db.transaction()`.
- **Status Transitions:** Implement strict state machine: `pending` -> `confirmed` -> `shipped` -> `delivered`; `pending` -> `cancelled`.
- **Auth:** All routes under `/api` require `x-api-key` header.
- **Timestamps:** Use `datetime('now')` for `created_at` and `updated_at`.
- **Categorization:** ID 1 is always reserved for "Uncategorized".
